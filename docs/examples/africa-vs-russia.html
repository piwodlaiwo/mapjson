Africa vs. Russia

On a Mercator map, Russia looks like the widest country on Earth and Africa looks modest by comparison — Mercator inflates east-west distances at high latitude, where meridians that actually converge toward the poles are drawn parallel instead. Measured for real (great-circle distance between each landmass's actual westernmost and easternmost point, computed client-side from the same geometry the map renders), Africa's east-west span is longer than Russia's. The two curves below are true geodesics on this projection, not straight lines — watch how the Russia line bows compared to the Africa one for the same real distance.

Russia's own shortest path from Kaliningrad to Chukotka runs within a couple of degrees of the North Pole, so the line is capped and drawn as two visible legs near each real endpoint rather than one continuous arc through the (off-frame) middle. The map itself is rotated so its seam falls in open Pacific water instead of through Chukotka — the default Mercator centering would otherwise cut Russia in two at the antimeridian.

API Call

https://api.mapjson.com/v1/geo?layer=countries&detail=medium&properties=name,continent&format=geojson

Code

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <title>Africa vs. Russia</title>
  <script src="https://cdn.jsdelivr.net/npm/d3@7/dist/d3.min.js"></script>
  <style>
    body { margin: 0; background: #cfe0ea; }
    svg { width: 100%; height: 100vh; display: block; }
  </style>
</head>
<body>
  <svg id="map"></svg>
  <script>
    const url = "https://api.mapjson.com/v1/geo?layer=countries&detail=medium&properties=name,continent&format=geojson";

    // Great-circle distance (haversine), km
    function haversine(lon1, lat1, lon2, lat2) {
      const R = 6371.0088, rad = Math.PI / 180;
      const dphi = (lat2 - lat1) * rad, dlambda = (lon2 - lon1) * rad;
      const a = Math.sin(dphi / 2) ** 2 + Math.cos(lat1 * rad) * Math.cos(lat2 * rad) * Math.sin(dlambda / 2) ** 2;
      return 2 * R * Math.asin(Math.sqrt(a));
    }

    // Every [lon, lat] vertex in a Polygon/MultiPolygon
    function flatten(geometry) {
      const pts = [];
      (function rec(c, depth) { depth === 0 ? pts.push(c) : c.forEach((x) => rec(x, depth - 1)); })
        (geometry.coordinates, geometry.type === "Polygon" ? 2 : 3);
      return pts;
    }

    // Westmost/eastmost vertex across a set of features. antimeridian:true
    // shifts negative longitudes +360 first — Russia spans the date line, so a
    // naive min/max would (wrongly) call Kaliningrad its "east" point.
    function extent(features, antimeridian) {
      let west = null, east = null;
      for (const f of features) {
        for (const [lon, lat] of flatten(f.geometry)) {
          const x = antimeridian && lon < 0 ? lon + 360 : lon;
          if (!west || x < west.x) west = { lon, lat, x, name: f.properties.name };
          if (!east || x > east.x) east = { lon, lat, x, name: f.properties.name };
        }
      }
      return { west, east };
    }

    const width = 960, height = 500;
    const svg = d3.select("#map").attr("viewBox", `0 0 ${width} ${height}`);

    d3.json(url).then(fc => {
      const africa = fc.features.filter((f) => f.properties.continent === "Africa");
      const russia = fc.features.filter((f) => f.properties.name === "Russia");
      const fit = fc.features.filter((f) => f.properties.name !== "Antarctica"); // Mercator + poles don't mix

      // Default centering puts the seam at the antimeridian, which cuts
      // straight through Chukotka (and the Aleutians, Fiji...). Rotating the
      // map moves that seam into open Pacific water instead, so Russia (and
      // everything else near the date line) renders as one unbroken shape.
      const projection = d3.geoMercator().rotate([-150, 0]).fitExtent([[8, 8], [width - 8, height - 8]], { type: "FeatureCollection", features: fit });
      const path = d3.geoPath(projection);

      const africaExtent = extent(africa, false);
      const russiaExtent = extent(russia, true);
      const africaKm = haversine(africaExtent.west.lon, africaExtent.west.lat, africaExtent.east.lon, africaExtent.east.lat);
      const russiaKm = haversine(russiaExtent.west.lon, russiaExtent.west.lat, russiaExtent.east.lon, russiaExtent.east.lat);
      // africaKm ≈ 9,800 · russiaKm ≈ 6,600 — real distance, despite how it looks below

      // The true shortest path between two points ~65°N and ~150° of
      // longitude apart (Russia's own extremes) swings to within a couple
      // degrees of the Pole — geometrically correct, but Mercator's
      // y-coordinate runs to +/-infinity at 90°, so drawing that whole arc
      // means most of the line is off-frame. Cap it at MAX_LAT: drop the
      // near-pole middle and keep the two visible legs near each endpoint as
      // separate lines (path() still antimeridian-clips each leg on its own,
      // same as it does for the country polygons).
      const MAX_LAT = 78;
      function geodesic(a, b, color) {
        const interpolate = d3.geoInterpolate([a.lon, a.lat], [b.lon, b.lat]);
        const raw = d3.range(0, 1.001, 0.005).map(interpolate);
        const lines = [[]];
        let lastKept = -2;
        raw.forEach((p, i) => {
          if (Math.abs(p[1]) > MAX_LAT) return;
          if (i !== lastKept + 1) lines.push([]);
          lines[lines.length - 1].push(p);
          lastKept = i;
        });
        svg.append("path")
          .datum({ type: "MultiLineString", coordinates: lines.filter((l) => l.length > 1) })
          .attr("d", path).attr("fill", "none").attr("stroke", color).attr("stroke-width", 2);
      }

      svg.selectAll(".bg").data(fc.features).join("path")
        .attr("d", path).attr("fill", "#ddd8cc").attr("stroke", "#f0ece6").attr("stroke-width", 0.4);
      svg.selectAll(".africa").data(africa).join("path")
        .attr("d", path).attr("fill", "#5a8f6e").attr("stroke", "#3d6b52").attr("stroke-width", 0.5);
      svg.selectAll(".russia").data(russia).join("path")
        .attr("d", path).attr("fill", "#b0402f").attr("stroke", "#8a3020").attr("stroke-width", 0.5);

      geodesic(africaExtent.west, africaExtent.east, "#2f6b4c");
      geodesic(russiaExtent.west, russiaExtent.east, "#7a2818");

      console.log(`Africa: ${africaKm.toFixed(0)} km`, `Russia: ${russiaKm.toFixed(0)} km`);
    });
  </script>
</body>
</html>
